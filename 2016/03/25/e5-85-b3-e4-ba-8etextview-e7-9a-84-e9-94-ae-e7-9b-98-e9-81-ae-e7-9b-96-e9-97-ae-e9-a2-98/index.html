关于TextView的键盘遮盖问题

发表于 2016-03-25 | 分类于 Home ， iOS

第一种方法：使用TextView的Delegate方法来实现

[code lang=”objc”] //如果键盘盖过输入框上移界面 -(void)textFieldDidBeginEditing:(UITextField *)textField { CGFloat offset=self.view.height- (textField.y+textField.height+300); if (offset<=0) { [UIView animateWithDuration:0.3 animations:^{ CGRect frame=self.view.frame; frame.origin.y=offset; self.view.frame=frame; }]; } } -(void)textFieldDidEndEditing:(UITextField *)textField { [UIView animateWithDuration:0.3 animations:^{ CGRect frame=self.view.frame; frame.origin.y=0; self.view.frame=frame; }]; } [/code]

第二种方法：注册通知监听键盘的弹入弹出，从而改变视图的偏移量

[code lang=”objc”] #pragma mark - 返回该页面时刷新 -(void)viewWillAppear:(BOOL)animated { [super viewWillAppear:animated]; [self.tableView reloadData]; [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(keyboardWillShow:) name:UIKeyboardWillShowNotification object:nil]; [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(keyboardWillHide:) name:UIKeyboardWillHideNotification object:nil]; [self addLineView]; } #pragma mark - 页面销毁后移除通知 -(void)viewDidDisappear:(BOOL)animated{ [[NSNotificationCenter defaultCenter] removeObserver:self]; } #pragma mark - scrollView滚动 使 textView 处于可显示区域 //这个方法一定根据需要，如果需要点击就触发事件则需要在textview的didchange代理方法里面调用这个方法。 -(void)scrollToShow:(UITextView )textView { CGRect cursorRect = [textView caretRectForPosition:textView.selectedTextRange.end]; [textView setContentOffset:CGPointZero animated:YES]; cursorRect.origin.y += 8; CGRect scrollRect = [self.tableView convertRect:cursorRect fromView:textView]; [self.tableView scrollRectToVisible:scrollRect animated:YES]; } #pragma mark - 监听键盘的弹出和收起 - (void)keyboardWillShow:(NSNotification *)notification { //键盘的frame NSValue \keyboardBoundsValue = [[notification userInfo] objectForKey:UIKeyboardFrameEndUserInfoKey]; CGRect keyboardBounds = [keyboardBoundsValue CGRectValue]; //设置 scrollView 的内边距即可，这里只需要将底部高度设置为键盘高度即可 UIEdgeInsets ei = self.tableView.contentInset; ei.bottom = keyboardBounds.size.height; [self.tableView setContentInset:ei]; //调整滑动条距离窗口底边的距离 [self.tableView setScrollIndicatorInsets:ei]; UITextView *textview = (UITextView *)findFirstResponseOfView(self.tableView); if (textview.tag >= 2000 || textview.tag >= 3000) { [self scrollToShow:textview]; } } - (void)keyboardWillHide:(NSNotification *)notification { UIEdgeInsets ei = self.tableView.contentInset; ei.bottom = 0; [self.tableView setContentInset:ei]; [self.tableView setScrollIndicatorInsets:ei]; } [/code]
