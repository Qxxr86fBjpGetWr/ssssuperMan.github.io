NSMutableAttributedString的简单使用

发表于 2016-03-29 | 分类于 Home ， iOS

使用NSMutableAttributedString可以修改同一个Label中不同段文字的属性，之前有使用过NSMutableAttributedString来做图文混排表情键盘，但是那个时候理解不是特别深入，今天再次遇到，但是只是一些简单的用法。

[code lang=”objc”] //统一设置Label中的文字颜色和大小 -(void)settingLabelTextAttribute:(UILabel *)label LabelText:(NSString *)string{ NSMutableAttributedString *attributeString = [[NSMutableAttributedString alloc]initWithString:string]; if ([string hasPrefix:@”就”]) { [attributeString setAttributes:@{NSForegroundColorAttributeName : UIColorFromRGB(0x20bed9),NSFontAttributeName : [UIFont systemFontOfSize:12]} range:NSMakeRange(0, 4)]; //range:NSMakeRange的范围这块会比较容易出现问题，当更改的时候应该注意空格的问题 [attributeString setAttributes: @{NSForegroundColorAttributeName : UIColorFromRGB(0x646464),NSFontAttributeName : [UIFont systemFontOfSize:12]} range:NSMakeRange(5, string.length - 5)]; } label.attributedText = attributeString; } [/code]
